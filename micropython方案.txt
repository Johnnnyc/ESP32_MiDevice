# ESP32 米家设备 MicroPython 转换方案

## 一、转换可行性总结

原Arduino代码可以转换为MicroPython，但需要注意以下几点：
- BLE功能需要使用MicroPython的ubluetooth模块实现
- DHT11传感器可直接使用dht库
- GPIO控制可通过machine.Pin模块实现
- 米家协议需要手动移植和实现

## 二、环境准备

1. **下载ESP32 MicroPython固件**：
   - 从 https://micropython.org/download/esp32/ 下载最新固件
   - 或使用支持BLE的自定义固件

2. **烧录固件**：
   - 使用esptool.py烧录固件到ESP32
   ```
   esptool.py --chip esp32 --port COMx erase_flash
   esptool.py --chip esp32 --port COMx --baud 460800 write_flash -z 0x1000 esp32-idf4-2023xxxx-v1.20.x.bin
   ```

3. **上传工具**：
   - 使用Thonny IDE、ampy或WebREPL上传文件

## 三、代码结构设计

将采用模块化设计，分为以下文件：

1. **config.py** - 配置文件，包含所有常量和配置项
2. **mi_device.py** - 米家设备功能实现
3. **main.py** - 主程序，初始化和主循环

## 四、模块转换详细方案

### 1. 配置模块 (config.py)

将原代码中的所有宏定义和配置项移至config.py

### 2. 米家设备模块 (mi_device.py)

_IRQ_CENTRAL_DISCONNECT = const(2)
_IRQ_GATTS_WRITE = const(3)

class MiDevice:
    def __init__(self):
        # 初始化设备状态
        self.switch_states = [False] * SWITCH_COUNT
        self.temperature = 0.0
        self.humidity = 0.0
        self.battery = 100
        
        # 设备信息
        self.mac_address = ""
        self.device_id = ""
        
        # BLE相关
        self.ble = ubluetooth.BLE()
        self.ble.active(True)
        self.ble.irq(self._irq)
        
        # 连接状态
        self.device_connected = False
        self.conn_handle = None
        
        # 注册服务和特征
        self._register_services()
    
    def _irq(self, event, data):
        # 处理BLE事件
        if event == _IRQ_CENTRAL_CONNECT:
            conn_handle, _, _, = data
            self.conn_handle = conn_handle
            self.device_connected = True
            print("设备已连接")
        
        elif event == _IRQ_CENTRAL_DISCONNECT:
            conn_handle, _, _, = data
            self.device_connected = False
            self.conn_handle = None
            print("设备已断开连接")
            # 重新开始广播
            self.ble.gap_advertise(100, self._build_advertisement_data())
            print("重新开始广播")
        
        elif event == _IRQ_GATTS_WRITE:
            conn_handle, value_handle = data
            # 读取写入的值
            data = self.ble.gatts_read(self.char_handle)
            self._parse_command(data)
        
        elif event == _IRQ_GATTS_READ_REQUEST:
            conn_handle, value_handle = data
            # 准备状态报告数据
            status_data = self._build_status_report()
            self.ble.gatts_write(self.char_handle, status_data)

            self.ble.gatts_read_perm(self.char_handle)

        print("收到米家控制命令:", [hex(b) for b in data])

### 3. 主程序 (main.py)

```python
import time
import dht
import machine
from machine import Pin, Timer
from config import *
from mi_device import MiDevice

# 初始化DHT11传感器
dht_sensor = dht.DHT11(Pin(DHT_PIN))

# 初始化开关引脚
switch_pins = [Pin(pin, Pin.OUT, value=0) for pin in SWITCH_PINS]

# 初始化米家设备
mi_device = MiDevice()

# 读取传感器数据
def read_sensor_data():
    try:
        dht_sensor.measure()
        temperature = dht_sensor.temperature()
        humidity = dht_sensor.humidity()
        print(f"DHT11读取成功: 温度={temperature:.1f}°C, 湿度={humidity:.1f}%")
        return temperature, humidity
    except OSError as e:
        print("DHT11传感器读取失败:", e)
        return None, None

# 更新GPIO开关状态
def update_gpio_states(switch_states):
    for i, state in enumerate(switch_states):
        switch_pins[i].value(1 if state else 0)

# 传感器数据更新计时器回调
def sensor_update_timer(timer):
    global mi_device
    temperature, humidity = read_sensor_data()
    if temperature is not None and humidity is not None:
        mi_device.temperature = temperature
        mi_device.humidity = humidity
        # 模拟电量消耗
        mi_device.battery = max(0, mi_device.battery - 0.1)
        # 更新GPIO状态
        update_gpio_states(mi_device.switch_states)
        # 更新设备数据
        mi_device.update_device_data()

# 主程序设置
def setup():
    print("========================================")
    print("ESP32 米家传感器开关设备 (MicroPython)")
    print("可被青萍盲网关发现并通过米家系统控制")
    print("支持DHT11温湿度传感器和多个IO开关")
    print("========================================")
    
    # 启动BLE广播
    mi_device.start_advertising()
    
    # 初始化传感器更新计时器
    timer = Timer(0)
    timer.init(period=STATUS_REPORT_INTERVAL, mode=Timer.PERIODIC, callback=sensor_update_timer)
    
    print("初始化完成，系统已启动")

# 主循环
def main():
    setup()
    while True:
        # 这里可以添加其他需要在主循环中执行的代码
        time.sleep(0.1)

# 启动程序
if __name__ == "__main__":
    main()
```

## 五、注意事项和挑战

1. **BLE功能差异**：
   - MicroPython的ubluetooth模块与Arduino的BLE库API差异较大
   - 需要仔细实现米家协议的帧格式和通信逻辑

2. **内存限制**：
   - ESP32的MicroPython固件有内存限制，需要注意代码大小
   - 可以考虑优化数据结构和减少不必要的功能

3. **调试建议**：
   - 使用Thonny IDE进行开发和调试
   - 利用串口打印进行错误排查
   - 分步测试各个功能模块

## 六、上传和测试流程

1. 上传三个Python文件到ESP32
2. 重启ESP32，程序会自动从main.py开始执行
3. 使用米家APP或青萍盲网关搜索设备
4. 测试设备发现、连接和控制功能
